def preprocessed
  preprocessed_string = TextPreprocessor.new(to_s)
                                        .substitute_coreferences
                                        .remove_determiners
                                        .to_s
  Text.new(preprocessed_string)
end

def sentences_objects
  @sentences_objects ||= begin
    acc = []
    each_nlp_sentence do |sent|
      acc.push(Sentence.new(sent))
    end
    acc
  end
end
